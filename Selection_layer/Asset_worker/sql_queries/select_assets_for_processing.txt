SELECT
    id,
    symbol,
    pipe_status,
    exchange_status,
    last_update_ts
FROM asset_universe
WHERE exchange_status = 'listed'
  AND (
        pipe_status = 'new'
        OR (
            pipe_status = 'skip_fresh'
            AND (
                last_update_ts IS NULL
                OR last_update_ts < NOW() - INTERVAL {skip_fresh_recheck_days} DAY
            )
        )
        OR (
            pipe_status = 'tested'
            AND (
                last_update_ts IS NULL
                OR last_update_ts < NOW() - INTERVAL {tested_refresh_days} DAY
            )
        )
    )
ORDER BY
    CASE pipe_status
        WHEN 'new' THEN 1
        WHEN 'skip_fresh' THEN 2
        WHEN 'tested' THEN 3
        ELSE 4
    END,
    last_update_ts ASC,
    id ASC
LIMIT {batch_size};
